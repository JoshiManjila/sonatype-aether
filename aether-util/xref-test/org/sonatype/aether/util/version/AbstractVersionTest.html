11  import static org.junit.Assert.assertEquals;
12  
13  import org.sonatype.aether.version.Version;
14  
15  abstract class AbstractVersionTest
16  {
17  
18      protected static final int X_LT_Y = -1;
19  
20      protected static final int X_EQ_Y = 0;
21  
22      protected static final int X_GT_Y = 1;
23  
24      protected abstract Version newVersion( String version );
25  
26      protected void assertOrder( int expected, String version1, String version2 )
27      {
28          Version v1 = newVersion( version1 );
29          Version v2 = newVersion( version2 );
30  
31          if ( expected > 0 )
32          {
33              assertEquals( "expected " + v1 + " > " + v2, 1, Integer.signum( v1.compareTo( v2 ) ) );
34              assertEquals( "expected " + v2 + " < " + v1, -1, Integer.signum( v2.compareTo( v1 ) ) );
35              assertEquals( "expected " + v1 + " != " + v2, false, v1.equals( v2 ) );
36              assertEquals( "expected " + v2 + " != " + v1, false, v2.equals( v1 ) );
37          }
38          else if ( expected < 0 )
39          {
40              assertEquals( "expected " + v1 + " < " + v2, -1, Integer.signum( v1.compareTo( v2 ) ) );
41              assertEquals( "expected " + v2 + " > " + v1, 1, Integer.signum( v2.compareTo( v1 ) ) );
42              assertEquals( "expected " + v1 + " != " + v2, false, v1.equals( v2 ) );
43              assertEquals( "expected " + v2 + " != " + v1, false, v2.equals( v1 ) );
44          }
45          else
46          {
47              assertEquals( "expected " + v1 + " == " + v2, 0, v1.compareTo( v2 ) );
48              assertEquals( "expected " + v2 + " == " + v1, 0, v2.compareTo( v1 ) );
49              assertEquals( "expected " + v1 + " == " + v2, true, v1.equals( v2 ) );
50              assertEquals( "expected " + v2 + " == " + v1, true, v2.equals( v1 ) );
51              assertEquals( "expected #(" + v1 + ") == #(" + v1 + ")", v1.hashCode(), v2.hashCode() );
52          }
53      }
54  
55      protected void assertSequence( String... versions )
56      {
57          for ( int i = 0; i < versions.length - 1; i++ )
58          {
59              for ( int j = i + 1; j < versions.length; j++ )
60              {
61                  assertOrder( X_LT_Y, versions[i], versions[j] );
62              }
63          }
64      }
