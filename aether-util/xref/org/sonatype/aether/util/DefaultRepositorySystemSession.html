47   * @author Benjamin Bentmann

364     private <T> Map<String, T> toSafeMap( Map<?, ?> table, Class<T> valueType )
365     {
366         Map<String, T> map;
367         if ( table == null || table.isEmpty() )
368         {
369             map = new HashMap<String, T>();
370         }
371         else
372         {
373             map = new LinkedHashMap<String, T>();
374             for ( Object key : table.keySet() )
375             {
376                 if ( key instanceof String )
377                 {
378                     Object value = table.get( key );
379                     if ( valueType.isInstance( value ) )
380                     {
381                         map.put( key.toString(), valueType.cast( value ) );
382                     }
383                 }
384             }
385         }
386         return map;
387     }
