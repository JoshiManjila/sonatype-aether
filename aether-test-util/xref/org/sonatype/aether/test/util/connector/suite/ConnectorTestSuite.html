129         List<ArtifactUpload> artUps = ConnectorTestUtils.createTransfers( ArtifactUpload.class, count, tmpFile );
130         List<MetadataUpload> metaUps = ConnectorTestUtils.createTransfers( MetadataUpload.class, count, tmpFile );
131         List<ArtifactDownload> artDowns = ConnectorTestUtils.createTransfers( ArtifactDownload.class, count, null );
132         List<MetadataDownload> metaDowns = ConnectorTestUtils.createTransfers( MetadataDownload.class, count, null );
133 
134         // this should block until all transfers are done - racing condition, better way to test this?
135         connector.put( artUps, metaUps );
136         connector.get( artDowns, metaDowns );
137 
138         for ( int i = 0; i < count; i++ )

140             ArtifactUpload artUp = artUps.get( i );
141             MetadataUpload metaUp = metaUps.get( i );
142             ArtifactDownload artDown = artDowns.get( i );
143             MetadataDownload metaDown = metaDowns.get( i );
144 
145             assertTrue( Transfer.State.DONE.equals( artUp.getState() ) );
146             assertTrue( Transfer.State.DONE.equals( artDown.getState() ) );
147             assertTrue( Transfer.State.DONE.equals( metaUp.getState() ) );
148             assertTrue( Transfer.State.DONE.equals( metaDown.getState() ) );
149         }

154     @Test
155     public void testMkdirConcurrencyBug()
156         throws IOException, NoRepositoryConnectorException
157     {
158         RepositoryConnector connector = factory().newInstance( session, repository );
159         File artifactFile = TestFileUtils.createTempFile( "mkdirsBug0" );
160         File metadataFile = TestFileUtils.createTempFile( "mkdirsBug1" );
161 
162         int numTransfers = 2;
163 
164         ArtifactUpload[] artUps = new ArtifactUpload[numTransfers];
165         MetadataUpload[] metaUps = new MetadataUpload[numTransfers];
166 
167 
168         for ( int i = 0; i < numTransfers; i++ )
169         {
170             StubArtifact art = new StubArtifact( "testGroup", "testArtifact", "", "jar", i + "-test" );
171             StubMetadata meta =
172                 new StubMetadata( "testGroup", "testArtifact", i + "-test", "maven-metadata.xml",
173                                   Metadata.Nature.RELEASE_OR_SNAPSHOT );
174 
175             ArtifactUpload artUp = new ArtifactUpload( art, artifactFile );
176             MetadataUpload metaUp = new MetadataUpload( meta, metadataFile );
177 
178             artUps[i] = artUp;
179             metaUps[i] = metaUp;
180         }
181 
182         connector.put( Arrays.asList( artUps ), null );
183         connector.put( null, Arrays.asList( metaUps ) );
184 
185         File localRepo = session.getLocalRepository().getBasedir();
186 
187         StringBuilder localPath = new StringBuilder( localRepo.getAbsolutePath() );
188 
189         for ( int i = 0; i < 50; i++ )
190         {
191             localPath.append( "/d" );
192         }
193 
194         ArtifactDownload[] artDowns = new ArtifactDownload[numTransfers];
195         MetadataDownload[] metaDowns = new MetadataDownload[numTransfers];
196 
197         for ( int m = 0; m < 20; m++ )
198         {
199             for ( int i = 0; i < numTransfers; i++ )
200             {
201                 File artFile = new File( localPath.toString() + "/a" + i );
202                 File metaFile = new File( localPath.toString() + "/m" + i );
203 
204                 StubArtifact art = new StubArtifact( "testGroup", "testArtifact", "", "jar", i + "-test" );
205                 StubMetadata meta =
206                     new StubMetadata( "testGroup", "testArtifact", i + "-test", "maven-metadata.xml",
207                                       Metadata.Nature.RELEASE_OR_SNAPSHOT );
208 
209                 ArtifactDownload artDown =
210                     new ArtifactDownload( art, null, artFile, RepositoryPolicy.CHECKSUM_POLICY_FAIL );
211                 MetadataDownload metaDown =
212                     new MetadataDownload( meta, null, metaFile, RepositoryPolicy.CHECKSUM_POLICY_FAIL );
213 
214                 artDowns[i] = artDown;
215                 metaDowns[i] = metaDown;
216             }
217 
218             connector.get( Arrays.asList( artDowns ), Arrays.asList( metaDowns ) );
219 
220             for ( int j = 0; j < numTransfers; j++ )
221             {
222                 ArtifactDownload artDown = artDowns[j];
223                 MetadataDownload metaDown = metaDowns[j];
224 
225                 assertNull( "artifact download had exception: " + artDown.getException(), artDown.getException() );
226                 assertNull( "metadata download had exception: " + metaDown.getException(), metaDown.getException() );
227                 assertEquals( State.DONE, artDown.getState() );
228                 assertEquals( State.DONE, metaDown.getState() );
229             }
230 
231             TestFileUtils.delete( localRepo );
232         }
233 
234         connector.close();
235     }

278     @Test
279     public void testProgressEventsDataBuffer()
280         throws UnsupportedEncodingException, IOException, NoSuchAlgorithmException, NoRepositoryConnectorException
281     {
282         byte[] bytes = "These are the test contents.\n".getBytes( "UTF-8" );
283         int count = 120000;
284         MessageDigest digest = MessageDigest.getInstance( "SHA-1" );
285         for (int i = 0; i < count; i++) {
286             digest.update( bytes );
287         }
288         byte[] hash = digest.digest();
289 
290         File file = TestFileUtils.createTempFile( bytes, count );
291 
292         Artifact artifact = new StubArtifact( "gid:aid:ext:ver" );
293         ArtifactUpload upA = new ArtifactUpload( artifact, file );
294 
295         File dir = TestFileUtils.createTempDir( "con-test" );
296         File downAFile = new File( dir, "downA.file" );
297         downAFile.deleteOnExit();
298         ArtifactDownload downA = new ArtifactDownload( artifact, "", downAFile, RepositoryPolicy.CHECKSUM_POLICY_FAIL );
299 
300         Metadata metadata =
301             new StubMetadata( "gid", "aid", "ver", "maven-metadata.xml", Metadata.Nature.RELEASE_OR_SNAPSHOT );
302         MetadataUpload upM = new MetadataUpload( metadata, file );
303         File downMFile = new File( dir, "downM.file" );
304         downMFile.deleteOnExit();
305         MetadataDownload downM = new MetadataDownload( metadata, "", downMFile, RepositoryPolicy.CHECKSUM_POLICY_FAIL );
306 
307         DigestingTransferListener listener = new DigestingTransferListener();
308         session.setTransferListener( listener );
309 
310         RepositoryConnector connector = factory().newInstance( session, repository );
311         connector.put( Arrays.asList( upA ), null );
312         assertArrayEquals( hash, listener.getHash() );
313         listener.rewind();
314         connector.put( null, Arrays.asList( upM ) );
315         assertArrayEquals( hash, listener.getHash() );
316         listener.rewind();
317         connector.get( Arrays.asList( downA ), null );
318         assertArrayEquals( hash, listener.getHash() );
319         listener.rewind();
320         connector.get( null, Arrays.asList( downM ) );
321         assertArrayEquals( hash, listener.getHash() );
322         listener.rewind();
323 
324         connector.close();
325     }

331     private final class DigestingTransferListener
332         implements TransferListener
333     {
334 
335         private MessageDigest digest;
336 
337         private synchronized void initDigest()
338             throws NoSuchAlgorithmException
339         {
340             digest = MessageDigest.getInstance( "SHA-1" );
341         }
342 
343         public DigestingTransferListener()
344             throws NoSuchAlgorithmException
345         {
346             initDigest();
347         }
348 
349         public void rewind()
350             throws NoSuchAlgorithmException
351         {
352             initDigest();
353         }
354 
355         public void transferSucceeded( TransferEvent event )
356         {
357         }
358 
359         public void transferStarted( TransferEvent event )
360             throws TransferCancelledException
361         {
362         }
363 
364         public synchronized void transferProgressed( TransferEvent event )
365             throws TransferCancelledException
366         {
367             digest.update( event.getDataBuffer() );
368         }
369 
370         public void transferInitiated( TransferEvent event )
371             throws TransferCancelledException
372         {
373         }
374 
375         public void transferFailed( TransferEvent event )
376         {
377         }
378 
379         public void transferCorrupted( TransferEvent event )
380             throws TransferCancelledException
381         {
382         }
383 
384         public synchronized byte[] getHash()
385         {
386             return digest.digest();
387         }
388     }
