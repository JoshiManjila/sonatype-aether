16  /**
17   * The result of a metadata resolution request.
18   * 
19   * @author Benjamin Bentmann
20   * @see RepositorySystem#resolveMetadata(RepositorySystemSession, java.util.Collection)
21   */
22  public class MetadataResult
23  {
24  
25      private final MetadataRequest request;
26  
27      private Exception exception;
28  
29      private boolean updated;
30  
31      private Metadata metadata;
32  
33      /**
34       * Creates a new result for the specified request.
35       * 
36       * @param request The resolution request, must not be {@code null}.
37       */
38      public MetadataResult( MetadataRequest request )
39      {
40          if ( request == null )
41          {
42              throw new IllegalArgumentException( "metadata request has not been specified" );
43          }
44          this.request = request;
45      }
46  
47      /**
48       * Gets the resolution request that was made.
49       * 
50       * @return The resolution request, never {@code null}.
51       */
52      public MetadataRequest getRequest()
53      {
54          return request;
55      }
56  
57      /**
58       * Gets the resolved metadata (if any).
59       * 
60       * @return The resolved metadata or {@code null} if the resolution failed.
61       */
62      public Metadata getMetadata()
63      {
64          return metadata;
65      }
66  
67      /**
68       * Sets the resolved metadata.
69       * 
70       * @param metadata The resolved metadata, may be {@code null} if the resolution failed.
71       * @return This result for chaining, never {@code null}.
72       */
73      public MetadataResult setMetadata( Metadata metadata )
74      {
75          this.metadata = metadata;
76          return this;
77      }
78  
79      /**
80       * Records the specified exception while resolving the metadata.
81       * 
82       * @param exception The exception to record, may be {@code null}.
83       * @return This result for chaining, never {@code null}.
84       */
85      public MetadataResult setException( Exception exception )
86      {
87          this.exception = exception;
88          return this;
89      }
90  
91      /**
92       * Gets the exception that occurred while resolving the metadata.
93       * 
94       * @return The exception that occurred or {@code null} if none.
95       */
96      public Exception getException()
97      {
98          return exception;
99      }
100 
101     /**
102      * Sets the updated flag for the metadata.
103      * 
104      * @param updated {@code true} if the metadata was actually fetched from the remote repository during the
105      *            resolution, {@code false} if the metadata was resolved from a locally cached copy.
106      * @return This result for chaining, never {@code null}.
107      */
108     public MetadataResult setUpdated( boolean updated )
109     {
110         this.updated = updated;
111         return this;
112     }
113 
114     /**
115      * Indicates whether the metadata was actually fetched from the remote repository or resolved from the local cache.
116      * If metadata has been locally cached during a previous resolution request and this local copy is still up-to-date
117      * according to the remote repository's update policy, no remote access is made.
118      * 
119      * @return {@code true} if the metadata was actually fetched from the remote repository during the resolution,
120      *         {@code false} if the metadata was resolved from a locally cached copy.
121      */
122     public boolean isUpdated()
123     {
124         return updated;
125     }
126 
127     /**
128      * Indicates whether the requested metadata was resolved. Note that the metadata might have been successfully
129      * resolved (from the local cache) despite {@link #getException()} indicating a transfer error while trying to
130      * refetch the metadata from the remote repository.
131      * 
132      * @return {@code true} if the metadata was resolved, {@code false} otherwise.
133      * @see Metadata#getFile()
134      */
135     public boolean isResolved()
136     {
137         return getMetadata() != null && getMetadata().getFile() != null;
138     }
139 
140     /**
141      * Indicates whether the requested metadata is not present in the remote repository.
142      * 
143      * @return {@code true} if the metadata is not present in the remote repository, {@code false} otherwise.
144      */
145     public boolean isMissing()
146     {
147         return getException() instanceof MetadataNotFoundException;
148     }
149 
150     @Override
151     public String toString()
152     {
153         return getMetadata() + ( isUpdated() ? " (updated)" : " (cached)" );
154     }
