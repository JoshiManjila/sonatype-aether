55      /**
56       * Expands a version range to a list of matching versions, in ascending order. For example, resolves "[3.8,4.0)" to
57       * ["3.8", "3.8.1", "3.8.2"].
58       * 
59       * @param session The repository session, must not be {@code null}.
60       * @param request The version range request, must not be {@code null}
61       * @return The version range result, never {@code null}.
62       * @throws VersionRangeResolutionException If the requested range could not be parsed. Note that an empty range does
63       *             not raise an exception.
64       */
65      VersionRangeResult resolveVersionRange( RepositorySystemSession session, VersionRangeRequest request )
66          throws VersionRangeResolutionException;

69       * Resolves an artifact's meta version (if any) to a concrete version. For example, resolves "1.0-SNAPSHOT" to
70       * "1.0-20090208.132618-23".
71       * 
72       * @param session The repository session, must not be {@code null}.
73       * @param request The version request, must not be {@code null}
74       * @return The version result, never {@code null}.
75       * @throws VersionResolutionException If the metaversion could not be resolved.

77      VersionResult resolveVersion( RepositorySystemSession session, VersionRequest request )

93      /**
94       * Collects the transitive dependencies of an artifact and builds a dependency graph. Note that this operation is

96       * artifact files, use {@link #resolveDependencies(RepositorySystemSession, DependencyRequest)}.
97       * 
98       * @param session The repository session, must not be {@code null}.
99       * @param request The collection request, must not be {@code null}
100      * @return The collection result, never {@code null}.
101      * @throws DependencyCollectionException If the dependency tree could not be built.
102      * @see RepositorySystemSession#getDependencyTraverser()
103      * @see RepositorySystemSession#getDependencyManager()
104      * @see RepositorySystemSession#getDependencySelector()
105      * @see RepositorySystemSession#getDependencyGraphTransformer()
106      */
107     CollectResult collectDependencies( RepositorySystemSession session, CollectRequest request )
108         throws DependencyCollectionException;

110     /**
111      * Collects and resolves the transitive dependencies of an artifact. This operation is essentially a combination of
112      * {@link #collectDependencies(RepositorySystemSession, CollectRequest)} and
113      * {@link #resolveArtifacts(RepositorySystemSession, Collection)}.
114      * 
115      * @param session The repository session, must not be {@code null}.
116      * @param request The dependency request, must not be {@code null}
117      * @return The dependency result, never {@code null}.
118      * @throws DependencyResolutionException If the dependency tree could not be built or any dependency artifact could

121     DependencyResult resolveDependencies( RepositorySystemSession session, DependencyRequest request )
122         throws DependencyResolutionException;

124     /**

126      * updated to reflect each successfully resolved artifact. Artifacts will be downloaded if necessary. Artifacts that

129      * @param session The repository session, must not be {@code null}.

133      * @return The resolution results, never {@code null}.
134      * @throws ArtifactResolutionException If any artifact could not be resolved.

176     ArtifactResult resolveArtifact( RepositorySystemSession session, ArtifactRequest request )

179     /**
180      * Resolves the paths for a collection of artifacts. Artifacts will be downloaded if necessary. Artifacts that are
181      * already resolved will be skipped and are not re-resolved. Note that this method assumes that any relocations have
182      * already been processed.

184      * @param session The repository session, must not be {@code null}.
