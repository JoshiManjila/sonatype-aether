15  /**
16   * A factory to create repository connectors. A repository connector is responsible for uploads/downloads to/from a
17   * certain kind of remote repository. When the repository system needs a repository connector for a given remote
18   * repository, it iterates the registered factories in descending order of their priority and calls
19   * {@link #newInstance(RepositorySystemSession, RemoteRepository)} on them. The first connector returned by a factory
20   * will then be used for the transfer.
21   * 
22   * @author Benjamin Bentmann
23   */
24  public interface RepositoryConnectorFactory
25  {
26  
27      /**
28       * Tries to create a repository connector for the specified remote repository. Typically, a factory will inspect
29       * {@link RemoteRepository#getProtocol()} and {@link RemoteRepository#getContentType()} to determine whether it can
30       * handle a repository.
31       * 
32       * @param session The repository system session from which to configure the connector, must not be {@code null}. In
33       *            particular, a connector must notify any {@link RepositorySystemSession#getTransferListener()} set for
34       *            the session and should obey the timeouts configured for the session.
35       * @param repository The remote repository to create a connector for, must not be {@code null}.
36       * @return The connector for the given repository, never {@code null}.
37       * @throws NoRepositoryConnectorException If the factory cannot create a connector for the specified remote
38       *             repository.
39       */
40      RepositoryConnector newInstance( RepositorySystemSession session, RemoteRepository repository )
41          throws NoRepositoryConnectorException;
42  
43      /**
44       * The priority of this factory. Factories with higher priority are preferred over those with lower priority.
45       * 
46       * @return The priority of this factory.
47       */
48      int getPriority();
49  
50  }
