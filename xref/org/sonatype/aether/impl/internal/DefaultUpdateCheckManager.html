82      public String getEffectiveUpdatePolicy( RepositorySystemSession session, String policy1, String policy2 )
83      {
84          return ordinalOfUpdatePolicy( policy1 ) < ordinalOfUpdatePolicy( policy2 ) ? policy1 : policy2;
85      }

87      private int ordinalOfUpdatePolicy( String policy )
88      {
89          if ( RepositoryPolicy.UPDATE_POLICY_DAILY.equals( policy ) )
90          {
91              return 1440;
92          }
93          else if ( RepositoryPolicy.UPDATE_POLICY_ALWAYS.equals( policy ) )
94          {
95              return 0;
96          }
97          else if ( policy != null && policy.startsWith( RepositoryPolicy.UPDATE_POLICY_INTERVAL ) )
98          {
99              String s = policy.substring( RepositoryPolicy.UPDATE_POLICY_INTERVAL.length() + 1 );
100             return Integer.valueOf( s );
101         }
102         else
103         {
104             // assume "never"
105             return Integer.MAX_VALUE;
106         }
107     }

109     public void checkArtifact( RepositorySystemSession session, UpdateCheck<Artifact, ArtifactTransferException> check )
110     {
111         if ( check.getLocalLastUpdated() != 0
112             && !isUpdatedRequired( session, check.getLocalLastUpdated(), check.getPolicy() ) )
113         {
114             if ( logger.isDebugEnabled() )
115             {
116                 logger.debug( "Skipped remote update check for " + check.getItem()
117                     + ", locally installed artifact up-to-date." );
118             }
119 
120             check.setRequired( false );
121             return;
122         }
123 
124         Artifact artifact = check.getItem();
125         RemoteRepository repository = check.getRepository();
126 
127         File artifactFile = check.getFile();
128         if ( artifactFile == null )
129         {
130             throw new IllegalArgumentException( String.format( "The artifact '%s' has no file attached", artifact ) );
131         }
132 
133         boolean fileExists = check.isFileValid() && artifactFile.exists();
134 
135         File touchFile = getTouchFile( artifact, artifactFile );
136         Properties props = read( touchFile );
137 
138         String updateKey = getUpdateKey( artifactFile, repository );
139         String dataKey = getDataKey( artifact, artifactFile, repository );
140 
141         String error = getError( props, dataKey );
142 
143         long lastUpdated;
144         if ( fileExists )
145         {
146             lastUpdated = artifactFile.lastModified();
147         }
148         else if ( error == null )
149         {
150             // this is the first attempt ever
151             lastUpdated = 0;
152         }
153         else if ( error.length() <= 0 )
154         {
155             // artifact did not exist
156             lastUpdated = getLastUpdated( props, dataKey );
157         }
158         else
159         {
160             // artifact could not be transferred
161             String transferKey = getTransferKey( artifact, artifactFile, repository );
162             lastUpdated = getLastUpdated( props, transferKey );
163         }
164 
165         if ( isAlreadyUpdated( session.getData(), updateKey ) )
166         {
167             if ( logger.isDebugEnabled() )
168             {
169                 logger.debug( "Skipped remote update check for " + check.getItem()
170                     + ", already updated during this session." );
171             }
172 
173             check.setRequired( false );
174             if ( error != null )
175             {
176                 check.setException( newException( error, artifact, repository ) );
177             }
178         }
179         else if ( lastUpdated == 0 )
180         {
181             check.setRequired( true );
182         }
183         else if ( isUpdatedRequired( session, lastUpdated, check.getPolicy() ) )
184         {
185             check.setRequired( true );
186         }
187         else if ( fileExists )
188         {
189             if ( logger.isDebugEnabled() )
190             {
191                 logger.debug( "Skipped remote update check for " + check.getItem()
192                     + ", locally cached artifact up-to-date." );
193             }
194 
195             check.setRequired( false );
196         }
197         else
198         {
199             if ( error == null || error.length() <= 0 )
200             {
201                 if ( session.isNotFoundCachingEnabled() )
202                 {
203                     check.setRequired( false );
204                     check.setException( newException( error, artifact, repository ) );
205                 }
206                 else
207                 {
208                     check.setRequired( true );
209                 }
210             }
211             else
212             {
213                 if ( session.isTransferErrorCachingEnabled() )
214                 {
215                     check.setRequired( false );
216                     check.setException( newException( error, artifact, repository ) );
217                 }
218                 else
219                 {
220                     check.setRequired( true );
221                 }
222             }
223         }
224     }

226     private ArtifactTransferException newException( String error, Artifact artifact, RemoteRepository repository )
227     {
228         if ( error == null || error.length() <= 0 )
229         {
230             return new ArtifactNotFoundException( artifact, repository, "Failure to find " + artifact + " in "
231                 + repository.getUrl() + " was cached in the local repository, "
232                 + "resolution will not be reattempted until the update interval of " + repository.getId()
233                 + " has elapsed or updates are forced" );
234         }
235         else
236         {
237             return new ArtifactTransferException( artifact, repository, "Failure to transfer " + artifact + " from "
238                 + repository.getUrl() + " was cached in the local repository, "
239                 + "resolution will not be reattempted until the update interval of " + repository.getId()
240                 + " has elapsed or updates are forced. Original error: " + error );
241         }
242     }

376     private long getLastUpdated( Properties props, String key )
377     {
378         String value = props.getProperty( key + UPDATED_KEY_SUFFIX, "" );
379         try
380         {
381             return ( value.length() > 0 ) ? Long.parseLong( value ) : 1;
382         }
383         catch ( NumberFormatException e )
384         {
385             logger.debug( "Cannot parse lastUpdated date: \'" + value + "\'. Ignoring.", e );
386             return 1;
387         }

405     private String getDataKey( Artifact artifact, File artifactFile, RemoteRepository repository )
406     {
407         Set<String> mirroredUrls = Collections.emptySet();
408         if ( repository.isRepositoryManager() )
409         {
410             mirroredUrls = new TreeSet<String>();
411             for ( RemoteRepository mirroredRepository : repository.getMirroredRepositories() )
412             {
413                 mirroredUrls.add( normalizeRepoUrl( mirroredRepository.getUrl() ) );
414             }
415         }
416 
417         StringBuilder buffer = new StringBuilder( 1024 );
418 
419         buffer.append( normalizeRepoUrl( repository.getUrl() ) );
420         for ( String mirroredUrl : mirroredUrls )
421         {
422             buffer.append( '+' ).append( mirroredUrl );
423         }
424 
425         return buffer.toString();
426     }

443     private String getRepoKey( RemoteRepository repository )
444     {
445         StringBuilder buffer = new StringBuilder( 128 );
446 
447         Proxy proxy = repository.getProxy();
448         if ( proxy != null )
449         {
450             appendAuth( buffer, proxy.getAuthentication() );
451             buffer.append( proxy.getHost() ).append( ':' ).append( proxy.getPort() ).append( '>' );
452         }
453 
454         Authentication auth = repository.getAuthentication();
455         appendAuth( buffer, auth );
456 
457         buffer.append( repository.getContentType() ).append( '-' );
458         buffer.append( normalizeRepoUrl( repository.getUrl() ) );
459 
460         return buffer.toString();
461     }

463     private String normalizeRepoUrl( String url )
464     {
465         String result = url;
466         if ( url != null && !url.endsWith( "/" ) )
467         {
468             result = url + '/';
469         }
470         return result;
471     }

473     private void appendAuth( StringBuilder buffer, Authentication auth )
474     {
475         if ( auth != null )
476         {
477             SimpleDigest digest = new SimpleDigest();
478             digest.update( auth.getUsername() );
479             digest.update( auth.getPassword() );
480             digest.update( auth.getPrivateKeyFile() );
481             digest.update( auth.getPassphrase() );
482             buffer.append( digest.digest() ).append( '@' );
483         }
484     }

491     private boolean isAlreadyUpdated( SessionData data, Object updateKey )
492     {
493         Object checkedFiles = data.get( SESSION_CHECKS );
494         if ( !( checkedFiles instanceof Map ) )
495         {
496             return false;
497         }
498         return ( (Map<?, ?>) checkedFiles ).containsKey( updateKey );
499     }

501     @SuppressWarnings( "unchecked" )
502     private void setUpdated( SessionData data, Object updateKey )
503     {
504         Object checkedFiles = data.get( SESSION_CHECKS );
505         while ( !( checkedFiles instanceof Map ) )
506         {
507             Object old = checkedFiles;
508             checkedFiles = new ConcurrentHashMap<Object, Object>( 256 );
509             if ( data.set( SESSION_CHECKS, old, checkedFiles ) )
510             {
511                 break;
512             }
513             checkedFiles = data.get( SESSION_CHECKS );
514         }
515         ( (Map<Object, Boolean>) checkedFiles ).put( updateKey, Boolean.TRUE );
516     }

518     public boolean isUpdatedRequired( RepositorySystemSession session, long lastModified, String policy )
519     {
520         boolean checkForUpdates;
521 
522         if ( policy == null )
523         {
524             policy = "";
525         }
526 
527         if ( RepositoryPolicy.UPDATE_POLICY_ALWAYS.equals( policy ) )
528         {
529             checkForUpdates = true;
530         }
531         else if ( RepositoryPolicy.UPDATE_POLICY_DAILY.equals( policy ) )
532         {
533             Calendar cal = Calendar.getInstance();
534             cal.set( Calendar.HOUR_OF_DAY, 0 );
535             cal.set( Calendar.MINUTE, 0 );
536             cal.set( Calendar.SECOND, 0 );
537             cal.set( Calendar.MILLISECOND, 0 );
538 
539             checkForUpdates = cal.getTimeInMillis() > lastModified;
540         }
541         else if ( policy.startsWith( RepositoryPolicy.UPDATE_POLICY_INTERVAL ) )
542         {
543             int minutes;
544             try
545             {
546                 String s = policy.substring( RepositoryPolicy.UPDATE_POLICY_INTERVAL.length() + 1 );
547                 minutes = Integer.valueOf( s );
548             }
549             catch ( RuntimeException e )
550             {
551                 minutes = 24 * 60;
552 
553                 logger.warn( "Non-parseable repository update policy '" + policy + "', assuming '"
554                     + RepositoryPolicy.UPDATE_POLICY_INTERVAL + ":1440'" );
555             }
556 
557             Calendar cal = Calendar.getInstance();
558             cal.add( Calendar.MINUTE, -minutes );
559 
560             checkForUpdates = cal.getTimeInMillis() > lastModified;
561         }
562         else
563         {
564             // assume "never"
565             checkForUpdates = false;
566 
567             if ( !RepositoryPolicy.UPDATE_POLICY_NEVER.equals( policy ) )
568             {
569                 logger.warn( "Unknown repository update policy '" + policy + "', assuming '"
570                     + RepositoryPolicy.UPDATE_POLICY_NEVER + "'" );
571             }
572         }
573 
574         return checkForUpdates;
575     }

583     public void touchArtifact( RepositorySystemSession session, UpdateCheck<Artifact, ArtifactTransferException> check )
584     {
585         Artifact artifact = check.getItem();
586         File artifactFile = check.getFile();
587         File touchFile = getTouchFile( artifact, artifactFile );
588 
589         String updateKey = getUpdateKey( artifactFile, check.getRepository() );
590         String dataKey = getDataKey( artifact, artifactFile, check.getAuthoritativeRepository() );
591         String transferKey = getTransferKey( artifact, artifactFile, check.getRepository() );
592 
593         setUpdated( session.getData(), updateKey );
594         Properties props = write( touchFile, dataKey, transferKey, check.getException() );
595 
596         if ( artifactFile.exists() && !hasErrors( props ) )
597         {
598             touchFile.delete();
599         }
600     }

628     private Properties write( File touchFile, String dataKey, String transferKey, Exception error )
629     {
630         Map<String, String> updates = new HashMap<String, String>();
631 
632         String timestamp = Long.toString( System.currentTimeMillis() );
633 
634         if ( error == null )
635         {
636             updates.put( dataKey + ERROR_KEY_SUFFIX, null );
637             updates.put( dataKey + UPDATED_KEY_SUFFIX, timestamp );
638             updates.put( transferKey + UPDATED_KEY_SUFFIX, null );
639         }
640         else if ( error instanceof ArtifactNotFoundException || error instanceof MetadataNotFoundException )
641         {
642             updates.put( dataKey + ERROR_KEY_SUFFIX, NOT_FOUND );
643             updates.put( dataKey + UPDATED_KEY_SUFFIX, timestamp );
644             updates.put( transferKey + UPDATED_KEY_SUFFIX, null );
645         }
646         else
647         {
648             String msg = error.getMessage();
649             if ( msg == null || msg.length() <= 0 )
650             {
651                 msg = error.getClass().getSimpleName();
652             }
653             updates.put( dataKey + ERROR_KEY_SUFFIX, msg );
654             updates.put( dataKey + UPDATED_KEY_SUFFIX, null );
655             updates.put( transferKey + UPDATED_KEY_SUFFIX, timestamp );
656         }
657 
658         return new TrackingFileManager().setLogger( logger ).update( touchFile, updates );
659     }
