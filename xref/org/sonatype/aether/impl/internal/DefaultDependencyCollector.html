133     public CollectResult collectDependencies( RepositorySystemSession session, CollectRequest request )
134         throws DependencyCollectionException
135     {
136         session = optimizeSession( session );
137 
138         RequestTrace trace = DefaultRequestTrace.newChild( request.getTrace(), request );
139 
140         CollectResult result = new CollectResult( request );
141 
142         DependencySelector depSelector = session.getDependencySelector();
143         DependencyManager depManager = session.getDependencyManager();
144         DependencyTraverser depTraverser = session.getDependencyTraverser();
145 
146         Dependency root = request.getRoot();
147         List<RemoteRepository> repositories = request.getRepositories();
148         List<Dependency> dependencies = request.getDependencies();
149         List<Dependency> managedDependencies = request.getManagedDependencies();
150 
151         GraphEdge edge = null;
152         if ( root != null )
153         {
154             VersionRangeResult rangeResult;
155             try
156             {
157                 VersionRangeRequest rangeRequest =
158                     new VersionRangeRequest( root.getArtifact(), request.getRepositories(), request.getRequestContext() );
159                 rangeRequest.setTrace( trace );
160                 rangeResult = versionRangeResolver.resolveVersionRange( session, rangeRequest );
161 
162                 if ( rangeResult.getVersions().isEmpty() )
163                 {
164                     throw new VersionRangeResolutionException( rangeResult, "No versions available for "
165                         + root.getArtifact() + " within specified range" );
166                 }
167             }
168             catch ( VersionRangeResolutionException e )
169             {
170                 result.addException( e );
171                 throw new DependencyCollectionException( result );
172             }
173 
174             Version version = rangeResult.getVersions().get( rangeResult.getVersions().size() - 1 );
175             root = root.setArtifact( root.getArtifact().setVersion( version.toString() ) );
176 
177             ArtifactDescriptorResult descriptorResult;
178             try
179             {
180                 ArtifactDescriptorRequest descriptorRequest = new ArtifactDescriptorRequest();
181                 descriptorRequest.setArtifact( root.getArtifact() );
182                 descriptorRequest.setRepositories( request.getRepositories() );
183                 descriptorRequest.setRequestContext( request.getRequestContext() );
184                 descriptorRequest.setTrace( trace );
185                 if ( isLackingDescriptor( root.getArtifact() ) )
186                 {
187                     descriptorResult = new ArtifactDescriptorResult( descriptorRequest );
188                 }
189                 else
190                 {
191                     descriptorResult = descriptorReader.readArtifactDescriptor( session, descriptorRequest );
192                 }
193             }
194             catch ( ArtifactDescriptorException e )
195             {
196                 result.addException( e );
197                 throw new DependencyCollectionException( result );
198             }
199 
200             root = root.setArtifact( descriptorResult.getArtifact() );
201 
202             repositories =
203                 remoteRepositoryManager.aggregateRepositories( session, repositories,
204                                                                descriptorResult.getRepositories(), true );
205             dependencies = mergeDeps( dependencies, descriptorResult.getDependencies() );
206             managedDependencies = mergeDeps( managedDependencies, descriptorResult.getManagedDependencies() );
207 
208             GraphNode node = new GraphNode();
209             node.setAliases( descriptorResult.getAliases() );
210             node.setRepositories( request.getRepositories() );
211 
212             edge = new GraphEdge( node );
213             edge.setDependency( root );
214             edge.setRequestContext( request.getRequestContext() );
215             edge.setRelocations( descriptorResult.getRelocations() );
216             edge.setVersionConstraint( rangeResult.getVersionConstraint() );
217             edge.setVersion( version );
218         }
219         else
220         {
221             edge = new GraphEdge( new GraphNode() );
222         }
223 
224         result.setRoot( edge );
225 
226         boolean traverse = ( root == null ) || depTraverser.traverseDependency( root );
227 
228         if ( traverse && !dependencies.isEmpty() )
229         {
230             DataPool pool = new DataPool( session );
231 
232             EdgeStack edges = new EdgeStack();
233             edges.push( edge );
234 
235             DefaultDependencyCollectionContext context =
236                 new DefaultDependencyCollectionContext( session, root, managedDependencies );
237 
238             Args args = new Args( result, session, trace, pool, edges, context );
239 
240             process( args, dependencies, repositories, depSelector.deriveChildSelector( context ),
241                      depManager.deriveChildManager( context ), depTraverser.deriveChildTraverser( context ) );
242         }
243 
244         DependencyGraphTransformer transformer = session.getDependencyGraphTransformer();
245         try
246         {
247             DefaultDependencyGraphTransformationContext context =
248                 new DefaultDependencyGraphTransformationContext( session );
249             result.setRoot( transformer.transformGraph( edge, context ) );
250         }
251         catch ( RepositoryException e )
252         {
253             result.addException( e );
254         }
255 
256         if ( !result.getExceptions().isEmpty() )
257         {
258             throw new DependencyCollectionException( result );
259         }
260 
261         return result;
262     }

271     private List<Dependency> mergeDeps( List<Dependency> dominant, List<Dependency> recessive )
272     {
273         List<Dependency> result;
274         if ( dominant == null || dominant.isEmpty() )
275         {
276             result = recessive;
277         }
278         else if ( recessive == null || recessive.isEmpty() )
279         {
280             result = dominant;
281         }
282         else
283         {
284             result = new ArrayList<Dependency>( dominant.size() + recessive.size() );
285             Collection<String> ids = new HashSet<String>();
286             for ( Dependency dependency : dominant )
287             {
288                 ids.add( getId( dependency.getArtifact() ) );
289                 result.add( dependency );
290             }
291             for ( Dependency dependency : recessive )
292             {
293                 if ( !ids.contains( getId( dependency.getArtifact() ) ) )
294                 {
295                     result.add( dependency );
296                 }
297             }
298         }
299         return result;
300     }

302     private String getId( Artifact a )
303     {
304         return a.getGroupId() + ':' + a.getArtifactId() + ':' + a.getClassifier() + ':' + a.getExtension();

307     private void process( final Args args, List<Dependency> dependencies, List<RemoteRepository> repositories,
308                           DependencySelector depSelector, DependencyManager depManager, DependencyTraverser depTraverser )
309         throws DependencyCollectionException
310     {
311         nextDependency: for ( Dependency dependency : dependencies )
312         {
313             boolean disableVersionManagement = false;
314 
315             List<Artifact> relocations = Collections.emptyList();
316 
317             thisDependency: while ( true )
318             {
319                 if ( !depSelector.selectDependency( dependency ) )
320                 {
321                     continue nextDependency;
322                 }
323 
324                 DependencyManagement depMngt = depManager.manageDependency( dependency );
325                 String premanagedVersion = null;
326                 String premanagedScope = null;
327 
328                 if ( depMngt != null )
329                 {
330                     if ( depMngt.getVersion() != null && !disableVersionManagement )
331                     {
332                         Artifact artifact = dependency.getArtifact();
333                         premanagedVersion = artifact.getVersion();
334                         dependency = dependency.setArtifact( artifact.setVersion( depMngt.getVersion() ) );
335                     }
336                     if ( depMngt.getProperties() != null )
337                     {
338                         Artifact artifact = dependency.getArtifact();
339                         dependency = dependency.setArtifact( artifact.setProperties( depMngt.getProperties() ) );
340                     }
341                     if ( depMngt.getScope() != null )
342                     {
343                         premanagedScope = dependency.getScope();
344                         dependency = dependency.setScope( depMngt.getScope() );
345                     }
346                     if ( depMngt.getExclusions() != null )
347                     {
348                         dependency = dependency.setExclusions( depMngt.getExclusions() );
349                     }
350                 }
351                 disableVersionManagement = false;
352 
353                 boolean noDescriptor = isLackingDescriptor( dependency.getArtifact() );
354 
355                 boolean traverse = !noDescriptor && depTraverser.traverseDependency( dependency );
356 
357                 VersionRangeResult rangeResult;
358                 try
359                 {
360                     VersionRangeRequest rangeRequest = new VersionRangeRequest();
361                     rangeRequest.setArtifact( dependency.getArtifact() );
362                     rangeRequest.setRepositories( repositories );
363                     rangeRequest.setRequestContext( args.result.getRequest().getRequestContext() );
364                     rangeRequest.setTrace( args.trace );
365 
366                     Object key = args.pool.toKey( rangeRequest );
367                     rangeResult = args.pool.getConstraint( key, rangeRequest );
368                     if ( rangeResult == null )
369                     {
370                         rangeResult = versionRangeResolver.resolveVersionRange( args.session, rangeRequest );
371                         args.pool.putConstraint( key, rangeResult );
372                     }
373 
374                     if ( rangeResult.getVersions().isEmpty() )
375                     {
376                         throw new VersionRangeResolutionException( rangeResult, "No versions available for "
377                             + dependency.getArtifact() + " within specified range" );
378                     }
379                 }
380                 catch ( VersionRangeResolutionException e )
381                 {
382                     addException( args.result, e );
383                     continue nextDependency;
384                 }
385 
386                 List<Version> versions = rangeResult.getVersions();
387                 for ( Version version : versions )
388                 {
389                     Artifact originalArtifact = dependency.getArtifact().setVersion( version.toString() );
390                     Dependency d = dependency.setArtifact( originalArtifact );
391 
392                     ArtifactDescriptorResult descriptorResult;
393                     {
394                         ArtifactDescriptorRequest descriptorRequest = new ArtifactDescriptorRequest();
395                         descriptorRequest.setArtifact( d.getArtifact() );
396                         descriptorRequest.setRepositories( repositories );
397                         descriptorRequest.setRequestContext( args.result.getRequest().getRequestContext() );
398                         descriptorRequest.setTrace( args.trace );
399 
400                         if ( noDescriptor )
401                         {
402                             descriptorResult = new ArtifactDescriptorResult( descriptorRequest );
403                         }
404                         else
405                         {
406                             Object key = args.pool.toKey( descriptorRequest );
407                             descriptorResult = args.pool.getDescriptor( key, descriptorRequest );
408                             if ( descriptorResult == null )
409                             {
410                                 try
411                                 {
412                                     descriptorResult =
413                                         descriptorReader.readArtifactDescriptor( args.session, descriptorRequest );
414                                     args.pool.putDescriptor( key, descriptorResult );
415                                 }
416                                 catch ( ArtifactDescriptorException e )
417                                 {
418                                     addException( args.result, e );
419                                     args.pool.putDescriptor( key, e );
420                                     continue;
421                                 }
422                             }
423                             else if ( descriptorResult == DataPool.NO_DESCRIPTOR )
424                             {
425                                 continue;
426                             }
427                         }
428                     }
429 
430                     d = d.setArtifact( descriptorResult.getArtifact() );
431 
432                     GraphNode node = args.edges.top().getTarget();
433 
434                     GraphEdge cycleEdge = args.edges.find( d.getArtifact() );
435                     if ( cycleEdge != null )
436                     {
437                         GraphEdge edge = new GraphEdge( cycleEdge.getTarget() );
438                         edge.setDependency( d );
439                         edge.setScope( d.getScope() );
440                         edge.setPremanagedScope( premanagedScope );
441                         edge.setPremanagedVersion( premanagedVersion );
442                         edge.setRelocations( relocations );
443                         edge.setVersionConstraint( rangeResult.getVersionConstraint() );
444                         edge.setVersion( version );
445                         edge.setRequestContext( args.result.getRequest().getRequestContext() );
446 
447                         node.getOutgoingEdges().add( edge );
448 
449                         continue;
450                     }
451 
452                     if ( !descriptorResult.getRelocations().isEmpty() )
453                     {
454                         relocations = descriptorResult.getRelocations();
455 
456                         disableVersionManagement =
457                             originalArtifact.getGroupId().equals( d.getArtifact().getGroupId() )
458                                 && originalArtifact.getArtifactId().equals( d.getArtifact().getArtifactId() );
459 
460                         dependency = d;
461                         continue thisDependency;
462                     }
463 
464                     d = args.pool.intern( d.setArtifact( args.pool.intern( d.getArtifact() ) ) );
465 
466                     DependencySelector childSelector = null;
467                     DependencyManager childManager = null;
468                     DependencyTraverser childTraverser = null;
469                     List<RemoteRepository> childRepos = null;
470                     Object key = null;
471 
472                     boolean recurse = traverse && !descriptorResult.getDependencies().isEmpty();
473                     if ( recurse )
474                     {
475                         DefaultDependencyCollectionContext context = args.collectionContext;
476                         context.set( d, descriptorResult.getManagedDependencies() );
477 
478                         childSelector = depSelector.deriveChildSelector( context );
479                         childManager = depManager.deriveChildManager( context );
480                         childTraverser = depTraverser.deriveChildTraverser( context );
481 
482                         childRepos =
483                             remoteRepositoryManager.aggregateRepositories( args.session, repositories,
484                                                                            descriptorResult.getRepositories(), true );
485 
486                         key =
487                             args.pool.toKey( d.getArtifact(), childRepos, childSelector, childManager, childTraverser );
488                     }
489                     else
490                     {
491                         key = args.pool.toKey( d.getArtifact(), repositories );
492                     }
493 
494                     List<RemoteRepository> repos;
495                     ArtifactRepository repo = rangeResult.getRepository( version );
496                     if ( repo instanceof RemoteRepository )
497                     {
498                         repos = Collections.singletonList( (RemoteRepository) repo );
499                     }
500                     else if ( repo == null )
501                     {
502                         repos = repositories;
503                     }
504                     else
505                     {
506                         repos = Collections.emptyList();
507                     }
508 
509                     GraphNode child = args.pool.getNode( key );
510                     if ( child == null )
511                     {
512                         child = new GraphNode();
513                         child.setAliases( descriptorResult.getAliases() );
514                         child.setRepositories( repos );
515 
516                         args.pool.putNode( key, child );
517                     }
518                     else
519                     {
520                         recurse = false;
521 
522                         if ( repos.size() < child.getRepositories().size() )
523                         {
524                             child.setRepositories( repos );
525                         }
526                     }
527 
528                     GraphEdge edge = new GraphEdge( child );
529                     edge.setDependency( d );
530                     edge.setScope( d.getScope() );
531                     edge.setPremanagedScope( premanagedScope );
532                     edge.setPremanagedVersion( premanagedVersion );
533                     edge.setRelocations( relocations );
534                     edge.setVersionConstraint( rangeResult.getVersionConstraint() );
535                     edge.setVersion( version );
536                     edge.setRequestContext( args.result.getRequest().getRequestContext() );
537 
538                     node.getOutgoingEdges().add( edge );
539 
540                     if ( recurse )
541                     {
542                         args.edges.push( edge );
543 
544                         process( args, descriptorResult.getDependencies(), childRepos, childSelector, childManager,
545                                  childTraverser );
546 
547                         args.edges.pop();
548                     }
549                 }
550 
551                 break;
552             }
553         }
554     }

556     private boolean isLackingDescriptor( Artifact artifact )
557     {
558         return artifact.getProperty( ArtifactProperties.LOCAL_PATH, null ) != null;
559     }

561     private void addException( CollectResult result, Exception e )
562     {
563         if ( result.getExceptions().size() < 100 )
564         {
565             result.addException( e );
